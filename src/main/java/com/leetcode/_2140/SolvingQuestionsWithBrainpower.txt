1. Создаем две переменных int n - длина входного массива, long[] dp = new long[n]
2. для dp[n-1]  = question[n-1][0];
3. итерируемся с конца -2 пока >=0
4. dp[i] = берем с входого массива с текущей строки 0 столбец (то есть по задаче берем количество получаемых пойнтов)
5. в переменную int skip берем с входого массива с текущей строки 1 столбец (то есть сколько элементов надо пропустить по задаче)
6. если (i + skip + 1 < n) То dp[i] += dp[i+skip+1];
7. В dp[i] берем максимальное из dp[i] (решаемого) и dp[i+1] (пропускаемого)
8. Вернуть dp[0];
